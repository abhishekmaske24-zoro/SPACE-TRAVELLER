// Three.js game implementation // Include necessary Three.js scripts import * as THREE from 'three'; // Game variables let scene, camera, renderer, score, gameOver, ship, enemies; const textures = { rocket: new THREE.TextureLoader().load('rocket.jpg'), asteroid: new THREE.TextureLoader().load('astroid.jpg') }; function init() { // Set up scene scene = new THREE.Scene(); camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); renderer = new THREE.WebGLRenderer(); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // Add ship ship = new THREE.Mesh(new THREE.BoxGeometry(1, 1, 1), new THREE.MeshBasicMaterial({ map: textures.rocket })); scene.add(ship); // Add enemies createEnemies(); // Set camera position camera.position.z = 5; score = 0; gameOver = false; animate(); } function createEnemies() { enemies = []; for (let i = 0; i < 5; i++) { const enemy = new THREE.Mesh(new THREE.SphereGeometry(0.5, 16, 16), new THREE.MeshBasicMaterial({ map: textures.asteroid })); enemy.position.set(Math.random() * 10 - 5, Math.random() * 10 - 5, Math.random() * 10 - 5); enemies.push(enemy); scene.add(enemy); } } function animate() { requestAnimationFrame(animate); if (!gameOver) { // Update game elements ship.rotation.x += 0.01; ship.rotation.y += 0.01; score += 1; // Track score // Render scene renderer.render(scene, camera); } else { showGameOverScreen(); } } function showGameOverScreen() { // Display game over screen alert('Game Over! Your score: ' + score); } function startGame() { init(); } // Touch controls for mobile window.addEventListener('touchstart', function() { if (gameOver) { // Restart game score = 0; gameOver = false; startGame(); } }); // Start the game startGame();
